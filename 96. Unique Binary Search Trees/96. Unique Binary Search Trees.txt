class Solution {
    public int numTrees(int n) {
        int[] dp = new int[n + 1];
        dp[0] = dp[1] = 1;
        int a;
        for(int i = 2; i <= n; i++){
            a = 0;
            for(int j = 1; j <= i; j++){
                a += dp[j-1] * dp[i - j];
            }
            dp[i] = a;
        }
        return dp[n];
    }
}
